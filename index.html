def show_menu():

def add_task():
    task = input("Enter task: ")
    tasks.append(task)
    print("Task added!")

def view_tasks():
    print("\n--- Your Tasks ---")
    for i, task in enumerate(tasks, 1):
        print(f"{i}. {task}")
    # Need to handle empty task list here

while True:
    show_menu()
    choice = input("Choose an option: ")

    if choice == '1':
        add_task()
    elif choice == '2':
        view_tasks()
    elif choice == '3':
        break
    else:
        print("Invalid choice. Try again.")
